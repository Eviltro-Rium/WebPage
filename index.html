// 月 M：埃拉托斯特尼筛法
int month() {
  const int N = 100;
  vector<bool> is_prime(N, true);
  is_prime[0] = is_prime[1] = false;
  for (int i = 2; i * i < N; ++i) {
    if (!is_prime[i]) continue;
    for (int j = i * i; j < N; j += i)
      is_prime[j] = false;
  }
  int cnt = 0;
  for (int i = 2; i < N; ++i) {
    if (!is_prime[i]) continue;
    if (++cnt == 5) return i;
  }
  return -1;
}
int M = month();

// 日 D：最小完全数
int first_perfect() {
  for (int n = 2; n < 32; ++n) {
    int s = 0;
    for (int k = 1; k < n; ++k)
      if (n % k == 0) s += k;
    if (s == n) return n;
  }
  return -1;
}
int D = first_perfect();
